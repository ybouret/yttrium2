\section{Quadratic}
Let $S$ be a local surface approximation
\begin{equation}
S(\vec{p},\vec\alpha) = a + bx + cy + d x^2 + e xy + f y^2
\end{equation}

Let $\vec p_1,\ldots,p_N$ be the approximating points with values $Z_1,\ldots,Z_N$:
\begin{equation}
D^2(\vec \alpha) = \frac{1}{2}\sum_{i=1}^N w_i^2 \left[Z_i - S(\vec p_i,\vec \alpha)\right]^2
\end{equation}

\begin{equation}
\left\lbrace
\begin{aligned}
	-\partial_a D^2 & = & \sum_{i=1}^N          \left[Z_i - S(\vec p_i,\vec \alpha)\right] w_i^2\\
    -\partial_b D^2 & = & \sum_{i=1}^N x_i      \left[Z_i - S(\vec p_i,\vec \alpha)\right] w_i^2\\
    -\partial_c D^2 & = & \sum_{i=1}^N y_i      \left[Z_i - S(\vec p_i,\vec \alpha)\right] w_i^2\\
    -\partial_d D^2 & = & \sum_{i=1}^N x_i^2    \left[Z_i - S(\vec p_i,\vec \alpha)\right] w_i^2\\
    -\partial_e D^2 & = & \sum_{i=1}^N x_i y_i  \left[Z_i - S(\vec p_i,\vec \alpha)\right] w_i^2\\
    -\partial_f D^2 & = & \sum_{i=1}^N y_i^2    \left[Z_i - S(\vec p_i,\vec \alpha)\right] w_i^2\\
\end{aligned}
\right.
\end{equation}

\begin{equation}
\begin{bmatrix}
\sum w_i^2  & \sum w_i^2 x_i & \sum w_i^2 y_i & \sum w_i^2 x_i^2 & \sum w_i^2 x_i y_i & \sum w_i^2 y_i^2\\
\end{bmatrix}
\cdot
\vec \alpha = 
\begin{bmatrix}
\sum w_i^2 Z_i\\
\sum w_i^2 x_i Z_i\\
\sum w_i^2 y_i Z_i\\
\sum w_i^2 x_i^2 Z_i\\
\sum w_i^2 x_i y _i Z_i\\
\sum w_i^2 y_i^2 Z_i\\
\end{bmatrix}
= 
\begin{bmatrix}
\overrightarrow {w^2} \\
\overrightarrow {w^2 x} \\
\overrightarrow {w^2 y} \\
\overrightarrow {w^2 x^2} \\
\overrightarrow {w^2 xy} \\
\overrightarrow {w^2 y^2} \\
\end{bmatrix}
\cdot \vec Z
\end{equation}

\section{PBR}

We assume that the intensity at  of a single point is:
\begin{equation}
	V_{ij}(\vec{r}) = I_{ij} \Psi\left( \vert \vec{r}-\vec{r}_{ij} \vert \right)
\end{equation}
\begin{equation}
	V(\vec{r}) = \sum_{i,j} I_{ij} \Psi\left( \vert \vec{r}-\vec{r}_{ij} \vert \right)
\end{equation}
where $\Psi$ is a decreasing function.
\begin{equation}
D^2(\vec \alpha) = \frac{1}{2}\sum_{i,j} \left[V(\vec{r}_{ij}) - S(\vec{r}_{ij},\vec \alpha)\right]^2
\end{equation}
